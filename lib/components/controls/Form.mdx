# Form
*A proxy from Formik `Form`*

# Field
*A Field for Form*

## Props
*Extends Formik `FormConfig`*

|Name|Type|Default|Description|
|----|----|-------|-----------|
|label|`string`|-|Label for the item|
|name|`string`|-|Name of the property that should be mapped to the input|
|children|`Element<Input>`|-|Input type element|

# FormConfig
*A proxy from Formik `FormConfig`*

## Props
*Extends Formik `FormConfig`*

|Name|Type|Default|Description|
|----|----|-------|-----------|
|initialValues|`?Values`|-|Initial values for the form|
|onSubmit|`(Values, FormikBag) => void`|-|Function to be called on submit|
|children|`formikBag => Element`|-|Content of the form|

## Usage
import {Playground} from "docz";

import Button from "../controls/Button"
import TextField from "../inputs/TextField"

import {FormConfig, Form, Field} from "./Form/index.js";

<Playground>
  <FormConfig
    initialValues={{count: 1}}
    onSubmit={console.log}
  >
    {({ isValid, handleSubmit }) => (
      <Form>
        <Field name="count" validate={value => Number(value) > 10 && "Should be less than 10"}>
          <TextField type="number"/>
        </Field>
        <Button bg="primary" c="light" isDisabled={!isValid} type="submit">Submit</Button>
      </Form>
    )}
  </FormConfig>
</Playground>
